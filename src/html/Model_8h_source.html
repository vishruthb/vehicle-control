197  AD<double> psides = CppAD::atan(coeffs_[1] + 2 * coeffs_[2] * x +

199

202  fxtut[Y] = y + v * CppAD::sin(psi) * dt_;
203  fxtut[PSI] = psi - v / Lf_ * delta * dt_;
204  fxtut[V] = v + a * dt_;
205  fxtut[CTE] = (f - y) + v * CppAD::sin(epsi) * dt_;
